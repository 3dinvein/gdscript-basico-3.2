CONDIÇÕES ( if / elif / else )

CONDIÇÃO IF


var vida = 100

func _ready():
     if vida == 100:
         print("igual")
pass

CONDIÇÃO ELIF


var vida = 100

func _ready():
     if vida == 100:
         print("igual")
     elif vida > 100:
         print("maior")
pass

CONDIÇÃO ELSE


var vida = 100

func _ready():
     if vida == 100:
         print("igual")
     elif vida > 100:
         print("maior")
     else:
         print("menor")
pass
